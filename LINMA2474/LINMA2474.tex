\chapter{Optimization on manifolds}
\section{Introduction}
Classical optimization methods like the gradient descent solve problems of the form 
\begin{equation}\label{eq:opt}
	\min_{x\in \M} f(x)
\end{equation}
for a set $M$. The methods rely on two key properties:
\begin{itemize}
	\item Linearity: $x_k$ and $\nabla f(x_k)$ belong to some vector space, in which they can be combined with linear operations;
	\item Inner product: $\nabla f(x_k)$ is the unique element of $\R^D$ such that 
	\begin{equation}
		\forall v\in \R^D, \: Df(x)[v] = \langle v, \nabla f(x)\rangle 
	\end{equation}
	where $Df(x)[v] = \lim_{t\to 0} \frac{f(x+tv) - f(x)}{t}$ is the directional derivative of $f$ at $x$ in the direction $v$.
\end{itemize}
There are two ways to see the problem \ref{eq:opt}: as a constrained optimization problem, or as an unconstrained optimization problem assuming that nothing else exists outside the set $\M$. Optimization on manifolds extends the classical unconstrained optimization algorithms to problems whose search space is a manifold (will be defined later).

\section{Definitions}
\subsection{Definition and properties of a manifold}
\begin{definition}[\textbf{Optimisation on manifolds}] 
	To minimize a function $f$ on a manifold $\M$, we need several objects:
	\begin{itemize}
		\item A \textcolor{orange}{local defining function} $h:\R^d\to \R$ such that $h^{-1}(0) = \M$;
		\item The \textcolor{green}{tangent space of $\M$} at some point $x$ is the local linearization of $\M$ at $x\in \M$: $u\in $tangent space of $\M$ at $x$ iff $u\in Ker(Dh(x))$;
		\item An inner product on the tangent spaces to define a new notion of gradient: \[Df(x)[v] = \langle \nabla f(x), v\rangle\]
		\item A \textcolor{purple}{retraction function}, i.e. a tool that allows to make a step on a manifold in a given tangent direction. 
	\end{itemize}
\end{definition}
\begin{definition}[\textbf{Smoothness}]
	$F:\mathcal{U}\subseteq \mathcal{E}\to \mathcal{V}\subseteq \mathcal{E}'$, with $\mathcal{U}, \mathcal{V}$ open, is said to be smooth if it is $\mathcal{C}^{\infty}$ on its domain.
\end{definition}
Let us explain the concepts needed for our optimization: 
\begin{definition}[\textbf{Embedded submanifold and \textcolor{orange}{local defining function}}]
	Let $\mathcal{E}$ be a linear space of dimension $d$. A non-empty subset $\M$ of $\mathcal{E}$ is a smooth embedded submanifold of $\mathcal{E}$ of dimension $n$ if either:
	\begin{itemize}
		\item $n=d$ and $\M$ is open in $\mathcal{E}$ (open submanifold);
		\item $n=d-k$ for some $k\ge 1$ and, for each $x\in \M$, there exists a neighbourhood $\mathcal{U}$ of $x$ in $\mathcal{E}$ and a smooth function $h:\mathcal{U}\to \R^k$. In that case, 
		\begin{itemize}
			\item $\M\cap \mathcal{U}=h^{-1}(0) = \{y\in \mathcal{U}:h(y)=0\}$ and 
			\item $rank(Dh(x)) = k$.
		\end{itemize}
	\end{itemize}
	Such a function $h$ is called a local defining function for $\M$ at $x$.
\end{definition}
\begin{definition}[\textbf{\textcolor{green}{Tangent space}}]
	Let $\M$ be a subset of $\mathcal{E}$. For all $x\in \M$, define 
	\begin{equation}
		\mathcal{T}_x\M = \{c'(0)\: | \: c:\mathcal{I}\to \M\text{ is smooth and }c(0)=x\}
	\end{equation}
	where $\mathcal{I}$ is any open interval containing $t=0$. That means that $v$ is in $\mathcal{T}_x\M$ iff there exists a smooth curve on $\M$ passing through $x$ with velocity $v$.
\end{definition}
Consider $\M$ an embedded submanifold of $\mathcal{E}$, $x\in \M$ and the set $\mathcal{T}_x\M$. 
\begin{itemize}
	\item If $\M$ is an open submanifold of $\mathcal{E}$, then $\mathcal{T}_x\M=\mathcal{E}$;
	\item Otherwise, $\mathcal{T}_x\M=Ker(Dh(x))$ with $h$ any local defining function at $x$. 
\end{itemize}
\begin{definition}[\textbf{Tangent bundle}]
	The tangent bundle is the set of all tangent spaces: $\mathcal{T}\M = \{(x,v):v\in \mathcal{T}_x\M\}$.
\end{definition}
\begin{definition}[\textbf{Map between manifolds}]
	Let $\M$ and $\M'$ be embedded submanifolds of $\mathcal{E}$ and $\mathcal{E}'$. A map $F:\M\to \M'$ is smooth iff $F=\bar F|_\M$ where $\bar F$ is some smooth map from a neighbourhood of $\M$ in $\mathcal{E}$ to $\mathcal{E}'$.
\end{definition}
\begin{definition}[\textbf{Differential of a map between manifolds}]
	The differential of $F:\M\to \M'$ at the point $x\in \M$ is the linear map $DF(x):\mathcal{T}_x\M\to \mathcal{T}_{F(x)}\M'$ defined by 
	\begin{equation}
		DF(x)[v]=\frac{d}{dt}F(c(t))|_{t=0} = (F\circ c)'(0)
	\end{equation}
	where $c$ is some smooth curve on $\M$ passing through $x$ at $t=0$ with velocity $v\in \mathcal{T}_x\M$.
	\begin{itemize}
		\item [$\to$] Note: the definition does not depend on the choice of the curve $c$: $DF(x) = D\bar F(x)|_{\mathcal{T}_x\M}$. 
	\end{itemize}
\end{definition}
\begin{definition}[\textbf{\textcolor{purple}{Retraction function}}]
	A retraction on a manifold $\M$ is a smooth map $R:\mathcal{TM}\to \M:(x,v)\to R_x(v)$ such that each curve $c(t)=R_x(tv)$ satisfies $c(0)=x$ and $c'(0)=v$.
	\begin{figure}[H]
		\centering
		\includegraphics[width=.5\textwidth]{img/retraction.png}
		\caption{Retraction on a sphere.}
		\label{fig:retraction}
	\end{figure}
\end{definition}
\subsection{Riemannian manifolds and metrics}
\begin{definition}[\textbf{Inner product}]
	As seen in previous courses, an inner product on $\mathcal{T}_x\M$ is a bilinear, symmetric, positive definite function $\langle \cdot, \cdot \rangle_x :\mathcal{T}_x\M \times \mathcal{T}_x\M\to \R$. Note that the inner product depends on the point of linearization. It induces some norm for tangent vectors: $\| u\|_x = \sqrt{\langle u,u\rangle_x}$. A metric on $\M$ is a choice of inner product $\langle \cdot,\cdot \rangle_x$ for each $\M$.
\end{definition}
\begin{definition}[\textbf{Metric}]
	A metric $x\to \langle \cdot,\cdot\rangle_x$ on $\M$ is a Riemannian metric if it varies smoothly with $x$, i.e. for all smooth vector fields $V,W$ on $\M$, the function $x\to \langle V(x), W(x)\rangle_x$ is smooth from $\M$ to $\R$. 
\end{definition}
\begin{definition}[\textbf{Riemannian manifold}]
	A Riemannian manifold is a manifold with a Riemannian metric. 
\end{definition}
\begin{definition}[\textbf{Riemannian distance}]
	Let $\M$ be a Riemannian manifold. Given a smooth curve $c:[a,b]\to \M$, we define the length of $c$ as 
	\begin{equation}
		L(c) = \int_a^b \| c'(t)\|_{c(t)}dt
	\end{equation}
	The Riemannian distance is then defined as $dist(x,y) = \inf_c L(c)$. 
\end{definition}
\begin{definition}[\textbf{Riemannian submanifolds}]
	Let $\M$ be an embedded submanifold of a Euclidean space $\mathcal{E}$. Equipped with the Riemannian metric obtained by restriction of the metric of $\mathcal{E}$, we call $\M$ a Riemannian submanifold of $\mathcal{E}$. 
\end{definition}
\subsection{Gradient on manifolds}
\begin{definition}[\textbf{Riemannian gradient}]
	Let $f:\M\to \R$ be smooth on a Riemannian manifold $\M$. The Riemannian gradient of $f$ is the vector field $\text{grad} f$ on $\M$ uniquely defined by the following identities:
	\begin{equation}
		\forall (x,v)\in \mathcal{T}\M, \qquad Df(x)[v] = \langle v, \text{grad}f(x)\rangle_x 
	\end{equation}
	where $Df(x)$ is the differential of $f$ and $\langle \cdot, \cdot \rangle_x$ is the Riemannian metric. 
\end{definition}
\begin{thm}
	Let $\M$ be a Riemannian submanifold of $\mathcal{E}$ endowed with the metric $\langle\cdot, \cdot\rangle$ and let $f:\M\to \R$ be a smooth function. The Riemannian gradient of $f$ is given by 
	\begin{equation}
		\text{grad}f(x) = \text{Proj}_x(\nabla \bar f(x))
	\end{equation}
	where $\bar f$ is any smooth extension of $f$ to a neighborhood of $\M$ in $\mathcal{E}$, and $\nabla \bar f(x)$ is the Euclidean gradient of $\bar f$ at $x$.
\end{thm}
\begin{itemize}
	\item [$\to$] Note: for $\mathcal{E} =\R^d$ and using the usual metric $\langle u,v\rangle = u^T v$, the projection operator is 
	\begin{equation}
		\text{Proj}_x(v) = v - (x^T v)x
	\end{equation}
\end{itemize}
\begin{proposition}
	Let $f:\M_1\to \M_2$ and $G:\M_2\to \M_3$ be smooth, where $\M_1, \M_2, \M_3$ are embedded submanifolds of $\mathcal{E}_1, \mathcal{E}_2, \mathcal{E}_3$ respectively. Then
	\begin{equation}
		G\circ F : \M_1\to \M_3:x\to G(F(x))
	\end{equation}
	is smooth and the chain rule applies:
	\begin{equation}
		D(G\circ F)(x)[v] = DG(F(x))[DF(x)[v]]
	\end{equation}
\end{proposition}
\subsection{Taylor development of functions defined on manifolds}
Let $f:\M\to \R$ be smooth and $c:\mathcal{I}\to \mathcal{M}$ be a smooth curve with $c(0)=x$ and $c'(0)=v$, with $\mathcal{I}\subseteq \R$ an open interval around $t=0$ and $\|v\|_x = 1$. Let us define $g:\mathcal{I}\to \R:t\to g(t) = f(c(t))$. Since $g=f\circ c$ is smooth and maps real numbers to real numbers, it admits a Taylor expansion:
\begin{equation}
	g(t) = g(0) + tg'(0) + \mathcal{O}(t^2)
\end{equation}
By the chain rule, 
\begin{equation}
	g'(t) = Df(c(t))[c'(t)] = \langle \text{grad}f(c(t)), c'(t)\rangle_{c(t)}
\end{equation}
and for $t=0$, 
\begin{equation}
	g(0) = f(x)\qquad g'(0) = \langle \text{grad}f(x), v\rangle_x
\end{equation}
Therefore, 
\begin{equation}
	\begin{aligned}
		f(c(t)) = f(x) + t\langle \text{grad}f(x), v\rangle_x + \mathcal{O}(t^2)\\
		f(R_x(tv)) = f(x) + \langle \text{grad}f(x), tv\rangle_x + \mathcal{O}(t^2)
	\end{aligned}
\end{equation}
And defining $s\coloneqq tv \in \mathcal{T}_x\M$, 
\begin{equation}
	f(R_x(s)) = f(x) + \langle \text{grad}f(x),s\rangle_x + \O(\|s\|_x^2)
\end{equation}
This allows to define the Riemannian gradient descent in the next chapter (see section \ref{sec:grad-descent}).
\section{Geometric considerations}
\subsection{Geodesics}
\begin{definition}[\textbf{Geodesics}]
	On a Riemannian manifold $\M$, a geodesic is a smooth curve $c:\mathcal{I}\to \M$ such that $c''(t)=0$ for all $t\in \mathcal{I}$, where $\mathcal{I}$ is an open interval of $\R$. The acceleration $c''(t)$ on a Riemannian submanifold is defined as follows:
	\begin{equation}
		c''(t) = \text{Proj}_{c(t)} (\ddot c(t)) \qquad \text{where}\qquad \ddot c(t) \coloneqq \frac{d^2}{dt^2}c(t)
	\end{equation}
	For example, the curve $c(t) = \cos(t\|v\|)x + \frac{\sin(t\|v\|)}{\|v\|} v$, where $x$ is the starting point and $v$ the initial velocity ($v\in \mathcal{T}_x\mathcal{S}^{d-1}$), i.e. the curve that traces a great circle on the sphere. 
	\begin{figure}[H]
		\centering 
		\includegraphics[width=.4\textwidth]{img/geodesic.png}
		\caption{Example of geodesic on the sphere.}
		\label{fig:geodesic}
	\end{figure}
\end{definition}
\begin{thm}
	Every geodesic $\gamma$ on $\M$ is locally minimizing, i.e. every $t$ in the domain of $\gamma$ has a neighborhood $\mathcal{I}$ in the domain of $\gamma$ such that, if $a,b\in \mathcal{I}$ with $a<b$, then the restriction $\gamma|_{[a,b]}$ is a minimizing curve. 
	\begin{itemize}
		\item [$\to$] Note: not all geodesics are minimizing, e.g. on the sphere, two points are always linked by two geodesics, a long one and a short one (red and black curve on figure \ref{fig:geodesic}).
	\end{itemize}
\end{thm}
\begin{thm}
	On a Riemannian manifold, for every $(x,v) \in \mathcal{TM}$, there exists a unique maximal geodesic $\gamma : \mathcal{I}\to \M$ with $\gamma(0)=x$ and $\gamma'(0)=v$, where maximal means that the interval $\mathcal{I}$ is as large as possible. This is equivalent to a Cauchy problem: 
	\begin{equation}
		\gamma''(t) = 0 \qquad \text{ such that }\begin{cases}
			\gamma(0) = x \\\gamma'(0)=v
		\end{cases}
	\end{equation}
\end{thm}
\begin{definition}[\textbf{Exponential map}]
	Consider the set \[\mathcal{D}=\{(x,v)\in \mathcal{TM} \: : \gamma \text{ is defined on an interval containing } [0,1]\}\] which is a subset of the tangent bundle. The exponential map $Exp:\mathcal{D}\to \M$ is defined by the relation $Exp(x,v) = \gamma(1)$.
	\begin{itemize}
		\item [$\to$] Note: the exponential map is a retraction. 
	\end{itemize}
\end{definition}
\subsection{Geodesic convexity}
As a reminder, in linear space, a set $\mathcal{S}$ is convex if, for all $x,y\in \mathcal{S}$, the line segment $t\to (1-t)x+ty$ for $t\in [0,1]$ is also in $\mathcal{S}$. A function $f:\mathcal{S}\to \R$ is a convex function if $\mathcal{S}$ is convex and for all $x,y\in \mathcal{S}$ we have 
\begin{equation}
	f((1-t)x+ty)\le (1-t)f(x)+tf(y)\qquad \forall t\in [0,1]
\end{equation}
\begin{definition}[\textbf{Geodesic convexity}]
	A subset $\mathcal{S}$ of a Riemannian manifold $\M$ is geodesically convex if, for every $x,y\in \mathcal{S}$, there exists a geodesic segment $c:[0,1]\to \M$ such that $c(0)=x$, $c(1)=y$ and $c(t)$ is in $\mathcal{S}$ for all $t\in [0,1]$. 
\end{definition}
\begin{definition}[\textbf{Geodesically convex function}]
	A function $f:\mathcal{S}\to \R$ is geodesically convex if $\mathcal{S}$ is geodesically convex and $f\circ c:[0,1]\to \R$ is convex for each geodesic segment $c:[0,1]\to \M$ whose image is in $\mathcal{S}$. 
\end{definition}
\begin{thm}
	Let $\mathcal{S}$ be a geodesically convex set on a Riemannian manifold $\M$ and let $f:\M\to \R$ be differentiable in a neighborhood of $\mathcal{S}$. Then, the restriction $f|_\mathcal{S}:\mathcal{S}\to \R$ is geodesically convex iff for all geodesic segments $c:[0,1]\to \M$ contained in $\mathcal{S}$, we have (for $x=c(0)$)
	\begin{equation}
		f(c(t)) \ge f(x) + t \langle \text{grad}f(x), c'(0)\rangle_x \qquad \forall t\in [0,1]
	\end{equation}
\end{thm}
\begin{thm}
	If $f$ is differentiable and geodesically convex on an open geodesically convex set, then $x$ is a global minimizer of $f$ iff $\text{grad}f(x)=0$. 
\end{thm}
\section{Euclidean and Non-Euclidean geometries}
In general, in geometry, we say that two lines are parallel iff they are disjoint. In Euclidean geometry, if $L$ is a line and if $P$ is a point not in $L$, then there exists one and only one line through $P$ that is parallel to $L$. However, in a hyperbolic geometry, there exist infinitely many distinct hyperbolic lines through $P$ that are parallel to $L$. Conversely, in elliptic geometries, there exist no line through $P$ that is parallel to $L$. \\

The curvature of a surface is defined by the biggest circle that stays on one side of the curve.

The directions in the normal plane where the curvature takes its maximum and minimum values ($k_1$ and $k_2$) are always perpendicular and are called principal directions. The product $k_1k_2$ of the two principal curvatures is called the Gaussian curvature.

\section{Topology tools}
\subsection{General topology}
\begin{definition}[\textbf{Topology}]
	A topology on a set $X$ is a collection $T$ of subsets of $X$, called open sets, such that 
	\begin{itemize}
		\item $X$ and $\emptyset$ belong to $X$;
		\item the union of the elements of any subcollection of $T$ is in $T$;
		\item the intersection of the elements of any finite subcollection of $T$ is in $T$.
	\end{itemize}
\end{definition}
\begin{definition}[\textbf{Hausdorff topological space.}]
	A topological space is a couple $(X,T)$ where $X$ is a set and $T$ is a topology on $X$. The topological space $(X,T)$ is Hausdorff if any two distinct points of $X$ have disjoint neighborhoods. If $X$ is Hausdorff, then every sequence of points of $X$ converges to at most one point in $X$. 
\end{definition}
\subsection{Topology for embedded submanifolds}
\begin{definition}[\textbf{Open set.}]
	A subset $\mathcal{U}$ of $\mathcal{M}$ is open (respectively closed) if it is the intersection of $\M$ with an open (respectively closed) set of $\E$.
\end{definition}
\begin{definition}[\textbf{Neighborhood.}]
	A neighborhood of $x$ in $\M$ is an open subset of $\M$ that contains $x$. \\
	A neighborhood of a subset of $\M$ is an open subset of $\M$ that contains that subset.
\end{definition}
\begin{definition}[\textbf{Limit.}]
	Consider a sequence $s$ of points $x_0,x_1,x_2,\dots$ on a manifold $\M$. 
	\begin{itemize}
		\item We say that a point $x$ in $\M$ is a limit of $s$ if, for all neighborhood $\mathcal{U}$ of $x$, there exists some $K\in \Z$ such that $x_k$ is in $\mathcal{U}$ for all $k\ge K$. \\
		The topology of a manifold is Hausdorff, hence a sequence has at most one limit. If $x$ is the limit, we write 
		\begin{equation}
			\lim_{k\to \infty} x_k = x \qquad \text{ or }\qquad x_k\to x
		\end{equation}
		and we say that the sequence converges to $x$.
		\item A point $x\in \M$ is an accumulation point of $s$ if it is the limit of a subsequence of $s$. 
	\end{itemize}
\end{definition}
\section{Riemannian gradient descent algorithm}\label{sec:grad-descent}
The basic algorithm is the following:
\begin{algorithm}[H]
	\label{algo:RGD}
	\caption{Riemannian gradient descent}
	\begin{algorithmic}[1]
		\State \textbf{Input:} $x_0\in \M$;
		\For{$k=0,1,2,\dots$}
			\State Pick a step size $\alpha_k >0$
			\State $x_{k+1} = R_{x_k}\left(-\alpha_k \text{grad} f(x_k)\right)$
		\EndFor	
	\end{algorithmic}
\end{algorithm}
There are several ways to choose the step size $\alpha_k$:
\begin{itemize}
	\item a fixed step size: $\alpha_k = \alpha$ for all $k$;
	\item optimal step size: compute $\alpha_k$ that minimizes exactly the function 
	\begin{equation}
		g(\alpha) = f\left(R_{x_k}(-\alpha \text{grad} f(x_k))\right)
	\end{equation}
	\item Backtracking: starting with a guess $\alpha_0>0$, iteratively reduce it by a factor $\tau \in (0,1)$ until it is deemed acceptable. 
\end{itemize}
\subsection{Global convergence}
Let $f:\M\to \R$ be a smooth function on a Riemannian manifold $\M$, and assume that 
\begin{itemize}
	\item there exists a lower bound $f_{low}\in \R$ such that $f(x)\ge f_{low}$ for all $x\in \M$;
	\item there exists a constant $c>0$ such that, for all $k\in \Z$, 
\end{itemize}
\begin{equation}\label{eq:decrease}
	f(x_k) - f(x_{k+1}) \ge c\| \text{grad} f(x_k)\|_{x_k}^2
\end{equation}
Then,
\begin{equation}
	\lim_{k\to \infty} \| \text{grad} f(x_k)\|_{x_k}^2 = 0 
\end{equation}
Furthermore, for all $K\ge1$, there exists $k\in \{0,\dots,K-1\}$ such that 
\begin{equation}
	\| \text{grad} f(x_k)\|_{x_k} \le \sqrt{\frac{f(x_0)-f_{low}}{cK}}
\end{equation}
This means that if a precision $\epsilon$ is required on the solution, the number of iterations that we will need has a complexity $\O(\epsilon^{-2})$. \\

The proof of these results is the same as done in the courses LINMA2471 or LINMA2460. \\

\subsection{Ensuring the assumptions}
In classical optimisation, the Lipschitz-smoothness assumption is that there exists a constant $L>0$ such that 
\begin{equation}
	\|\nabla f(x+s)-\nabla f(x)\| \le L\|s\| \qquad \forall x,s\in \R^d 
\end{equation}
implied by the expression 
\begin{equation}
	f(x+s) \le f(x) + \langle \nabla f(x),s\rangle  + \frac{L}{2}\|s\|^2 \qquad \forall x,s\in \R^d
\end{equation}
However, on manifolds, the gradient is defined differently and with a different support. We need to modify this condition. \\
On a manifold $\M$, for a given subset $S$ of the tangent bundle $\mathcal{TM}$, there exists a constant $L>0$ such that, forall $(x,s)\in S$, 
\begin{equation}
	f(R_x(s)) \le f(x) + \langle \text{grad} f(x),s\rangle + \frac{L}{2}\|s\|_x^2
\end{equation}
From this, we can ensure the assumption of equation \eqref{eq:decrease}: define $x\coloneqq x_k$, $s\coloneqq -\alpha_k \text{grad}f(x_k)$ and $x_{k+1} \coloneqq R_x(s)$. Then,
\begin{equation}
	f(x_{k+1}) \le f(x_k) - \alpha_k\left(1-\frac{L\alpha_k}{2}\right) \| \text{grad}f(x_k)\|_{x_k}^2
\end{equation}
and this gives 
\begin{equation}
	f(x_k) - f(x_{k+1}) \ge \alpha_k \left(1-\frac{L\alpha_k}{2}\right) \| \text{grad} f(x_k)\|_{x_k}^2 
\end{equation}
Therefore, the assumption is true taking $c = \alpha_k \left(1-\frac{L\alpha_k}{2}\right)$.
\subsection{Optimal step size}
From the previous proof, we know that the decrease of the iterate is best when the constant $c$ is maximized. We can thus define the function 
\begin{equation}
	g:\R\to \R : \alpha \to \alpha \left(1-\frac{L\alpha}{2}\right)
\end{equation}
which is maximized for $\alpha^* = 1/L$, the same value seen for optimization in $\R^d$. For that value, the constant is $c=1/2L$.
\subsection{Backtracking}
If the Lipschitz constant $L$ is unknown, a possible algorithm for the optimal step size if backtracking line-search using the Armijo-Goldstein condition. This condition is 
\begin{equation}
	f(x) - f(R_x(-\alpha \text{grad}f(x))) \ge r\alpha \|\text{grad}f(x)\|_x^2
\end{equation}
and the algorithm is 
\begin{algorithm}[H]
	\caption{Backtracking line-search}
	\begin{algorithmic}[1]
		\State \textbf{Parameters:} $\tau,r\in (0,1)$, e.g. $\tau=1/2$ and $r=10^{-4}$;
		\State \textbf{Input:} $x\in \M$ and $\bar \alpha >0$;
		\State Set $\alpha \leftarrow \bar \alpha$;
		\While{$f(x) - f(R_x(-\alpha \text{grad}f(x))) < r\alpha \|\text{grad}f(x)\|_x^2$}
		\State $\alpha \leftarrow \tau \alpha$
		\EndWhile \\
		\Return $\alpha$
	\end{algorithmic}
\end{algorithm}
Under an assumption similar to the Lipschitz condition on the gradient, a convergence rate of $\O(1/\epsilon^2)$ can still be guaranteed. \\

\section{Dynamical systems}
The goal is to analyse some high-dimensional system using low-dimensional models, using the hypothesis that some dimensions of the system are correlated. This chapter aims to explain some existing methods to find some intrisic low-dimensional structures. An example is the reconstruction of images, well-researched using SVD, where the columns of pixels are highly correlated and only a few allow to reconstruct the image quite well. 

This section aims to reduce high-dimensional dynamical systems to lower-dimensional ones, by applying some linear applications on the input and output of the system, allowing to use less observations. \\

Dynamical systems in physics can generally be modelled by a set of differential equations, giving a state-space model:
\begin{equation}
	\begin{aligned}
		\dot x(t) &= f(x(t),u(t))\\
		y(t) &= g(x(t),u(t))
	\end{aligned}
\end{equation}
where $x\in \R^n$ is the state, $u\in \R^{n_i}$ is the input and $y\in \R^{n_o}$ is the observed output. As their expressions are governed by the system, $x$ and $y$ cannot be free and are restricted to a low-dimensional submanifold in their respective functional space. 
\subsection{The discrete linear time invariant model}
This model is 
\begin{equation}
	\begin{aligned}
		x(t+1) &= Ax(t) +Bu(t)\\
		y(t) &= Cx(t)+Du(t)\\
	\end{aligned}
\end{equation}
Our goal is to determine the dimension $n$ of the state-space model, and the matrix parameters $A,B,C,D$.\\
For that, let us define two Hankel matrices: 
\begin{equation}
	\begin{aligned}	
		Y &= \begin{bmatrix}
			y(1) & y(2) & \dots & y(N)\\
			y(2) & y(3) & \dots & y(N+1)\\
			\vdots & \vdots & \ddots & \vdots \\
			y(N) & y(N+1) & \dots & y(2N-1)
		\end{bmatrix} \in \R^{n_oN\times N} \\ U &= \begin{bmatrix}
			u(1) & u(2) & \dots & u(N)\\
			u(2) & u(3) & \dots & u(N+1)\\
			\vdots & \vdots & \ddots & \vdots \\
			u(N) & u(N+1) & \dots & u(2N-1)
		\end{bmatrix} \in \R^{n_iN\times N}
	\end{aligned}
\end{equation}
From the system, we know that 
\begin{equation}
	\begin{aligned}
		y(1) &= Cx(1) + Du(1)\\
		y(2) &= Cx(2) + Du(2) = C(Ax(1) + Bu(1)) + Du(2)\\
		\vdots
	\end{aligned}
\end{equation}
By this recurrence, we can see that 
\begin{equation}
	\begin{bmatrix}
	y(1) \\\vdots \\ y(r)
	\end{bmatrix} = \underbrace{\begin{bmatrix} C\\ CA\\\vdots \\ CA^{r-1}\end{bmatrix}}_{\eqqcolon G}x(1) + \underbrace{\begin{bmatrix}
		D &&&&\\
		CB & D &&&\\
		CAB & CB & D && \\
		\vdots & \ddots & \ddots & \ddots & \\
		CA^{r-2}B & CA^{r-3}B & \dots & CB & D
	\end{bmatrix}}_{\eqqcolon H} \begin{bmatrix}
		u(1) \\ \vdots \\u(r)
	\end{bmatrix}
\end{equation}
And thus 
\begin{equation}
	Y = GX+HU
\end{equation}
where $X=[x(1), x(2),\dots,x(N)]\in \R^{n\times N}$. Let $U^\perp$ span the columns of the nullspace of $U$. Then, $UU^\perp=0$, and 
\begin{equation}
	YU^\perp = GXU^\perp + HUU^\perp = GXU^\perp
\end{equation}
We call $YU^\perp$ the input-output matrix, and this is always of rank lower than the dimension of the state space: $rank(YU^\perp)\le n$. \\
Man-made structures and object often display patterns and/or symmetries in their making, meaning that their actual information space has a much lower dimensionality. Therefore, using the matrix $YU^\perp$ is much more efficient. 
\subsection{Discretize a continuous system}
As numerical methods can only consider discrete-time systems and not continuous ones, we need a way to discretize the systems correctly, in order to capture as much information as the continuous formulation. The low-dimensional hypothesis allows to a relatively small number of samples because of the low rank of the modifies output $YU^T$. 

This comes from the Fourier transform of the signal $x(t)$:
\begin{equation}
    x(t) = \frac{1}{2\pi}\int_{-\infty}^\infty \hat x(\omega)\exp{(i\omega t)}d\omega = \frac{1}{2\pi} \int_{-\Omega}^\Omega \hat x(\omega) \exp{(i\omega t)}d\omega 
\end{equation}
If $\hat x(\omega)$ is viewed as a periodic function in the spectral domain with a period $2\Omega$, it is then fully determined by its Fourier coefficients: 
\begin{equation}
    x\left(\frac{n\pi}{\Omega}\right) = \frac{1}{2\pi}\int_{-\Omega}^{\Omega} \hat x(\omega) \exp{\left(i\omega \frac{n\pi}{\Omega}\right)}d\omega
\end{equation}
and $f=\frac{\Omega}{\pi}$ is the frequency of the sampling. 
\begin{itemize}
    \item [$\bullet$] Note: to perfectly recover a band-limited signal $x(t)$, we need to sample it at a rate that is twice its maximal frequency $f_{\max} = \Omega/2\pi$.
\end{itemize}

To discretize a function $x(t):\R\to \R$ to a function $x_T(t):\N\to \R$, we use the Sha function:
\begin{equation}
	\Sha(t) \coloneqq \sum_{k=-\infty}^\infty \delta (t-k)
\end{equation}
With that function, the discretization is straightforward:
\begin{equation}
	x_T(t) = x(t) \cdot \Sha\left(\frac{t}{T}\right)
\end{equation}
And in the frequency domain:
\begin{equation}
	\mathcal{F}(x_T(t)) = \hat x(\nu) \star T\Sha(t\nu)
\end{equation}

\subsection{Latent variables}
One may view the observed data as samples of a set of random variables $\boldsymbol{x}_o\in \R^{n_o}$ generated through certain conditional probability distribution given another set of hidden (said latent) variables $\boldsymbol{x}_h\in \R^{n_h}$. 

In that case, the structure of the data is fully described by the joint distribution of the vector $x=(x_o,x_h)\in \R^n$, for $n=n_o+n_h$. Assuming that $x\sim \mathcal{N}(0,\Sigma)$, where $\Sigma \in \R^{n\times n}$ is the covariance matrix, we define $\Theta=\Sigma^{-1}$ the precision matrix, or concentration matrix. In the case of jointly Gaussian variables, we can partition the covariance matrix as follows:
\begin{equation}
	\Sigma = \begin{bmatrix}
		\Sigma_o & \Sigma_{o,h}\\ \Sigma_{o,h}^T & \Sigma_h 
	\end{bmatrix} = \begin{bmatrix}
		\Theta_o & \Theta_{o,h}\\ \Theta_{o,h}^T & \Theta_h 
	\end{bmatrix}^{-1} \in \R^{n\times n}
\end{equation}
As the matrices of hidden variables are by definition not observed, we need a tool to connect $\Sigma_o$ to $\Theta_o$. This tool is the Schur complement:
\begin{equation}
	\Sigma_o^{-1} = \Theta_o - \Theta_{o,h}\Theta_h^{-1}\Theta_{o,h}^T \in \R^{n_o\times n_o}
\end{equation}
If the observed random variables are mainly independent, then $\Theta_o$ is sparse. Moreover, the second term has a rank less than the number of hidden variables $n_h$, which itself is often relatively small compared to $n_o$. Therefore, the covariance matrix $\Sigma_o$ of the observed variables is always a decomposable structure $\Sigma_o^{-1} = S+L$, where $S$ is sparse and $L$ is low-rank, two forms that are very useful in high-dimensional analysis, meaning that it can easily be represented, compared to a full matrix of size $n_o\times n_o$. \\
Earlier results allow us to write the conditional probabilities:\\

If $\begin{pmatrix}
	x_a\\x_b
\end{pmatrix} \sim \mathcal{N}\left(\begin{pmatrix}
	\mu_a\\\mu_b
\end{pmatrix}, \begin{pmatrix}
	\Sigma_a& \Sigma_{a,b}\\ \Sigma_{a,b} & \Sigma_b
\end{pmatrix}\right)$, then 
\begin{equation}
	x_a|x_b \sim \mathcal{N}(\bar \mu, \bar \Sigma) \qquad \begin{cases}
		\bar \mu = \mu_a + \Sigma_{a,b}\Sigma_b^{-1} (x_b-\mu_b)\\
		\bar \Sigma = \Sigma_a - \Sigma_{a,b}\Sigma_b^{-1}\Sigma_{a,b}^T
	\end{cases}
\end{equation}

\section{Sparse signal modeling}
Often, the number of unknowns vastly exceeds the number of observations: we have $y=Ax$, for $y\in \R^n$ the observations, $A\in \R^{m\times n}$ the data generation process and $x\in \R^n$ the unkowns. We consider $m\ll n$. The system thus have infinitely many solutions or none. To find the one solution that we want, we need to leverage some additional properties such as sparsity. \\
\subsection{Example in medical imaging}
Consider a digital $N\times N$ image $I\in \R^{N^2}$ (not the identity matrix!). This image, in an MRI, corresponds to the density of protons at a given spatial location in the brain. The input $u$ denotes how the magnetic field we applied varies over space. Denote $\mathcal{F}$ the 2D Fourier transform, we have
\begin{equation}\label{eq:irm}
	y = \mathcal{F}[I](u)
\end{equation}
By changing the applied magnetic field, $u$ varies and we can collect $m$ samples of the Fourier transform: $\{u_1,\dots,u_m\}$. This gives \eqref{eq:irm} in $m$ dimensions. The system seems impossible to solve, but we can use a known structure about it: the wavelet transform. 
\begin{equation}
	I = \Psi x
\end{equation}
where $\Psi$ is a matrix defined from $\psi\in L^2(\R)$, with 
\begin{equation}
	\Psi_{j,k}(x) \coloneqq 2^{j/2} \psi(2^jx-k)\qquad x\in \R
\end{equation}
The function $\psi$ is said to be a wavelet if the system $\{\Psi_{j,k}\}_{j,k\in \Z}$ is an orthonormal basis of $L^2(\R)$\footnote{The scalar product used is $\langle f,g\rangle \eqqcolon \int_{-\infty}^{\infty} f(x)g(x)dx$}. Moreover, $x\in \R^{N^2}$ containts the coefficients of the image $I$ with respect to the basis $\Psi$. Many of these coefficients are extremely small, meaning that $x$ is sparse. We can try to reconstruct this vector $x$ by using the observation equation:
\begin{equation}
	\begin{aligned}
		y = \mathcal{F}_U[I] &= \mathcal{F}_U\left[ \psi_1x_1+\dots + \psi_{N^2}x_{N^2}\right]\\
		&= \mathcal{F}_U[\psi_1]x_1 + \dots + \mathcal{F}_U(\psi_{N^2})x_{N^2}\\
		&= \begin{bmatrix} \mathcal{F}_U[\psi_1] \dots \mathcal{F}_U[\psi_{N^2}] x \end{bmatrix}\\
		&= Ax
	\end{aligned}
\end{equation}
for the matrix $A\in \R^{m\times N^2}$.\\
This system is much underdetermined, but we can use the sparsity of $x$ to solve it: we consider that only its $k$ largest coefficients are significant and the others are negligible. The solution $\hat x$ will be used to define an approximation of $I$:
\begin{equation}
	\hat I = \sum_{i=1}^{N^2} \psi_i \hat x_i
\end{equation}
\subsection{Recovering a sparse solution}
Recall the three properties of a norm in a space $V$:
\begin{itemize}
	\item Nonnegative homogeneity: $\|\alpha x\| = |\alpha| \|x\|$ for all vector $x\in V$ and scalars $\alpha\in \R$;
	\item Positive definiteness: $\|x\|\ge 0$ and $\|x\|=0\Longleftrightarrow x=0$;
	\item Subadditivity: $\|x+y\|\le \|x\| + \|y\|$ for all $x,y\in V$.
\end{itemize}
The $\ell_0$ norm is the number of nonzero entries in a vector $x$:
\begin{equation}
	\|x\|_0 = \#\{i|x(i)\neq 0\}
\end{equation}
It is thus a measure of the sparsity of a vector. It is not a formal norm, as it does not verify the nonnegative homogeneity property, but it can be seen as a limit of the $\ell^p$ norm:
\begin{equation}
	\|x\|_0 = \lim_{p\to 0^+} \|x\|_p^p
\end{equation}
\subsubsection{Minimizing the $\ell^0$ norm}
Knowing $y$ and $A$, we want to find $x_o$ by recovering the sparsest $x$ such that $Ax=y$. Namely, we want to find
\begin{equation}
	\hat x\in \arg\min \|x\|_0 \qquad \text{ s.t. } Ax=y
\end{equation}
\begin{thm}\label{thm:fuck}
	Assume that $y=Ax_o$, $\|x_o\|_0 \le k$, $\hat x$ exists and $\{\delta \in \R^n | A\delta = 0, \|\delta \|_0 \le 2k\}=\{0\}$, i.e. the nullspace of $A$ has no sparse vector. Then $\hat x=x_o$ and is the unique solution of the $\ell^0$ minimization. 
	\begin{itemize}
		\item [$\to$] Note: the bound on the sparsity of $\delta$ is $2k$ because the norm of $\hat x-x_o$ is bounded by $2k$: $\|\hat x-x_o\|_0 \le \|\hat x\|_0 + \|x_o\|_0 \le 2k$ and $A\delta = A(\hat x-x_o) = A\hat x-Ax_o = y-y=0$.
	\end{itemize}
\end{thm}
\begin{definition}
	The Kruskal rank is defined as 
	\begin{equation}
		krank(A) \coloneqq \max_r \{r\in \N | \text{ the columns of }A_I \text{ are linearly independent }\color{red}\forall\color{black} I\subset [n]\text{ with } |I| = r\}
	\end{equation}
\end{definition}
The difference with the usual rank is the red $\forall$, that is a $\exists$ for the usual rank. 
\begin{itemize}
	\item [$\to$] Note: $0\le krank(A)\le rank(A)\qquad \forall A$.
	\item [$\to$] Note: the nullspace assumption of theorem \ref{thm:fuck} is equivalent to saying $krank(A)\ge 2k$. 
\end{itemize}
\begin{thm}[\textbf{$\ell^0$ recovery.}]
	Suppose that $y=Ax_o$, with $\|x_o\|_0 \le \frac{1}{2}krank(A)$. Then, $x_o$ is the unique optimal solution to the $\ell^0$ minimization problem. 
\end{thm}
All these results make the $\ell^0$ minimization problem well-posed from an initial ill-posed problem, but the worst-case complexity of an algorithm solving this problem is $n^k$, for $k$ the number of nonzero entries that we want to recover: the problem is NP-hard.
\subsection{Relaxing the sparse recovery problem}
As the problem is NP-hard, we need a remplacement to make it easier to solve. We can see that in 1D, the absolute value function is the convex hull of the $\ell^0$ norm on the interval $[-1,1]$ (see figure \ref{fig:l1l0}).
\begin{figure}[H]
	\centering 
	\includegraphics[width=.6\textwidth]{img/l1l0.png}
	\caption{Replacing the $\ell^0$ norm by its convex envelope.}
	\label{fig:l1l0}
\end{figure}
Then, the $\ell^1$ norm provides a good replacement for the $\ell^0$ norm and simplifies the problem to a convex optimisation problem. 
\begin{equation}
	\min_x \|x\|_1 \qquad \text{ such that } y=Ax
\end{equation}
To solve this new problem, we can use a projected subgradient method: 
\begin{equation}
	x_{t+1} \coloneqq \mathcal{P}_\mathcal{C} \left(x_t - \frac{1}{t}sign(x_t)\right)
\end{equation}
for $\mathcal{P}_\mathcal{C}(z) = \arg \min_{x\in \mathcal{C}} \frac{1}{2}\|z-x\|^2_2$. Here, we use $\mathcal{C} = \{x:Ax=y\}$ and the sign function is the subgradient of the $\ell^1$ norm.\\ If $A$ has full row rank (plausible in the general case), then the projection is easy:
\begin{equation}
	\mathcal{P}_{\{x|Ax=y\}} (z) = z - A^T (AA^T)^{-1}(Az-y)
\end{equation}
The algorithm to minimize the $\ell_1$-norm on a constrained domain is thus 
\begin{algorithm}[H]
	\caption{$\ell_1$-minimization by projected subgradient}
	\begin{algorithmic}[1]
		\State Input: $A\in \R^{m\times n}$ and $y=\R^m$.
		\State Compute the projection operator $\Gamma \leftarrow I-A^T(AA^T)^{-1}A$ and the projection of $y$ $\bar x\leftarrow A^\dagger y = A^T(AA^T)^{-1}y$;
		\State $x_0\leftarrow 0$ and $t\leftarrow 0$;
		\While{Not converged}
			\State $t\leftarrow t+1$;
			\State $x_t \leftarrow \bar x + \Gamma(x_{t-1}-\frac{1}{t}sign(x_t))$
		\EndWhile
	\end{algorithmic}
\end{algorithm}
The $\ell_1$ minimization works as a replacement of the $\ell_0$ minimization, because the solution will always be on the vertices or the edges of the $\ell_1$ ball. Those correspond to sparse vectors because most components are zero on those elements of the ball. It can be seen as the following property of linear optimisation: in a polytope, the optimum of a linear function will always be a vertex or a complete edge of the polytope. There exists a mapping from the ball to a polytope (see figure \ref{fig:polytope}).

	\label{fig:polytope}
\end{figure}
\section{Convex method for sparse signal recovery}
\subsection{Incoherent matrices}
\begin{definition}
	The \textbf{mutual coherence} of a matrix $A = \left[ a_1 |\cdots| a_n \right] \in \R^{m \times n}$ with all nonzero columns is defined as:
	\begin{equation}
		\mu(A) = \max_{i \neq j} \left| \langle \frac{a_i}{\|a_i\|_2}, \frac{a_{\mathcal{J}}}{\|a_{\mathcal{J}}\|_2} \rangle \right|
	\end{equation}
\end{definition}
The mutual coherence is useful for sparse recovery because we will show that when $\mu(A)$ is small, the $krank(A)$ will be large. Namely, 
\begin{equation}
	krank(A) \ge \frac{1}{\mu(A)}.
\end{equation}
This is because if the columns of $A$ are close to orthogonal, then any subset of them will be linearly independent and it will be well-conditioned as their smallest singular value $\sigma_{\min}$ will be close to their largest singular value $\sigma_{\max}$. 
To prove it, consider $A_{\mathcal{J}}$ is the full-rank submatrix of $A$ formed by the columns indexed by $\mathcal{J}\subset [n]$, with $|\mathcal{J}| = k$. Let us write the matrix under a form that is the sum of the diagonal and off-diagonal ($\Delta$) entries:
\begin{equation}
	A_{\mathcal{J}}^* A_{\mathcal{J}} = I + \Delta
\end{equation}
We can bound the spectral norm of $\Delta$ by its Frobenius norm, then by the max norm, and finally by the mutual coherence:
\begin{equation}
	\|\Delta\|_{SP} \leq \|\Delta\|_F < k \|\Delta\|_{\infty} \leq k \mu(A)\\
\end{equation}
because
\begin{equation}
	\max_i \sigma_i(\Delta) \le \sqrt{\sum_i\sigma_i^2} = \sqrt{\sum_{ij}|\Delta_{ij}^2|} = \sqrt{\sum_{i\neq j} |\Delta_{ij}|^2} \le \sqrt{k(k-1) \|\Delta\|^2_\infty} < k \|\Delta\|_\infty
\end{equation}
Then using this result, we can bound the smallest singular value of $A_{\mathcal{J}}^*A_{\mathcal{J}}$. Let $\Delta = U D U^T$ be an eigenvalue decomposition then $\|\Delta\|_{SP} = \max_i |d_i|$:
\begin{equation}
	\begin{aligned}
		\sigma_{\min}(A_{\mathcal{J}}^* A_{\mathcal{J}}) &= \sigma_{\min}(I + \Delta) = \sigma_{\min}(U(I + D) U^T)\\
		&= \min_i |1 + d_i| \geq \min_i 1 - |d_i|\\ 
		&= 1 - \max_i |d_i| = 1 - \|\Delta\|_{SP} \\
		&> 1 - k \mu(A)
	\end{aligned}
\end{equation}
Then we have:
\begin{equation}\label{eq:label}
	1-k \mu(A) < \sigma_{\min}(A_{\mathcal{J}}^* A_{\mathcal{J}}) \leq \sigma_{\max}(A_{\mathcal{J}}^* A_{\mathcal{J}}) < 1 + k \mu(A)
\end{equation}
With the previous result, we can see that if $1 - k \mu(A) \geq 0$ or equivalently $k \leq 1/\mu(A)$, then $\sigma_{\min}(A_{\mathcal{J}}^* A_{\mathcal{J}}) > 0$ and thus $A_{\mathcal{J}}$ is full rank.\\
Combining these results with the definition of krank, we have the following result:
\begin{equation}
	krank(A) \geq \frac{1}{\mu(A)}
\end{equation}
This can be proven like this:\\
Let $s \coloneqq krank(A) + 1$. We normalize the columns of $A$, this does not change the $krank$ nor the mutual coherence.  Then $\exists a_{j_1}, \ldots, a_{j_s}$ linearly dependent, and thus $\alpha_1 a_{j_1} + \cdots + \alpha_s a_{j_s} = 0$ for some $\alpha_i \neq 0$ and $\alpha_1 = \max_{i} |\alpha_i|$. Then, we have
\begin{equation}
	\begin{aligned}
		0 &= \langle a_{j_1}, \alpha_1 a_{j_1} + \cdots + \alpha_s a_{j_s} \rangle\\
		\alpha_1 &= - \alpha_2 \langle a_{j_1}, a_{j_2} \rangle - \cdots - \alpha_s \langle a_{j_1}, a_{j_s} \rangle\\
		\frac{\alpha_1}{|\alpha_1|} &= \left( \frac{|\alpha_2|}{|\alpha_1|} + \cdots + \frac{|\alpha_s|}{|\alpha_1|} \right) \mu(A) \\
		1 &\leq (s-1) \mu(A)\\
		\frac{1}{\mu(A)} &\leq krank(A) 
	\end{aligned}
\end{equation}
With this result, we can see that if $y = Ax_o$ and $\|x_o\|_0 \leq \frac{1}{2 \mu(A)}$ then $x_o$ is the unique optimal solution to the $\ell^0$ minimization problem. 
\begin{equation}
	\min_x \|x\|_0 \qquad \text{ such that } Ax=y
\end{equation}
\begin{thm}[$\ell^1$ suceeds under incoherence]\label{thm:ell1_inco}
	Let $A \in \R^{m \times n}$ be an incoherent matrix with mutual coherence $\mu(A)$, with $diag(A^TA) = I$, $y = Ax_o$, and $\|x_o\|_0 \leq \frac{1}{2 \mu(A)}$. Then, $x_o$ is the unique optimal solution to the $\ell^1$ minimization problem.
\end{thm}
This theorem can be proven if the three following points are true. Let $x' \in \R^n$ feasible point $\neq 0$, $\mathcal{J} = supp(x_o)$, $k = |\mathcal{J}|$, we want:
\begin{enumerate}
	\item $\exists \lambda \in \R^m$ such that 
	\begin{equation}\label{eq:sign-cond}
		\begin{cases}
			A_\mathcal{J}^T \lambda = sign(x_{o,\mathcal{J}})\\
			\|A_{\mathcal{J}^c}^T \lambda\|_\infty < 1
		\end{cases}
	\end{equation}
	\item $\|x'\|_1 \geq \|x_o\|_1 + (1-\|A_{\mathcal{J}^c}^T \lambda\|_\infty) \|[x'-x_o]_{\mathcal{J}^c}\|_1$
	\item $[x'-x_o]_{\mathcal{J}^c} \neq 0$
\end{enumerate}
If these three points are true, then $\|x'\|_1 > \|x_o\|_1$ and thus $x_o$ is the unique optimal solution to the $\ell^1$ minimization problem. If conditions 1 and 3 are true, then $\|x'\|_1$ is strictly larger than $\|x_o\|_1$ by condition 2. \\

Let us prove each point:
\begin{enumerate}
	\item Let $\lambda = A_\mathcal{J} (A_\mathcal{J}^*A_\mathcal{J})^{-1} sign(x_{o,\mathcal{J}})$, then the first condition of \eqref{eq:sign-cond} is satisfied. For the second condition, we have for $j \in \mathcal{J}^c$, 
	\begin{equation}
		\begin{aligned}
			|a_{\mathcal{J}}^* A_\mathcal{J} (A_\mathcal{J}^*A_\mathcal{J})^{-1} sign(x_{o,\mathcal{J}})| &\leq \underbrace{\|A_\mathcal{J}^* a_{\mathcal{J}}\|_2}_{\le \mu(A)} \underbrace{\|(A_\mathcal{J}^*A_\mathcal{J})^{-1}\|_{SP}}_{\le \frac{1}{1-k\mu(A)} \text{by \eqref{eq:label}}} \underbrace{\|sign(x_{o,\mathcal{J}})\|_2}_{\le k}\\
			&\leq \frac{k \mu(A)}{1- k \mu(A)} \leq 1 
		\end{aligned}
	\end{equation}
	This proves the \eqref{eq:label} because if all elements of the vector are bounded by 1, then its max norm too. The last inequality is true because $k\mu(A) \leq 1/2$.
	\item Let $v \in \R^{n}$ defined by $v_\mathcal{J} = sign(x_{o,\mathcal{J}})$ and $v_{\mathcal{J}^c} = sign([x'-x_o]_{\mathcal{J}^c})$. Then if we observe that $v \in \partial \|\cdot\|_1 (x_o)$ and that by 1) $A^* \lambda \in \partial \|\cdot\|_1 (x_o)$, we can use the subgradient inequality:
	\begin{equation}
		\|x'\|_1 \ge \|x_o\|_1 + \langle v,x'-x_o\rangle
	\end{equation}
	Since $x'-x_o\in \text{null}(A)$, $\langle A^*\lambda, x'-x_o\rangle = 0$ and 
	\begin{equation}
		\begin{aligned}
			\|x'\|_1 &\geq \|x_o\|_1 + \langle v - A^* \lambda, x'-x_o\rangle\\
			&= \|x_o\|_1 + \langle [v - A^* \lambda]_{\mathcal{J}^c}, [x'-x_o]_{\mathcal{J}^c}\rangle\\
			&= \|x_o\|_1 + \langle v_{\mathcal{J}^c}, [x'-x_o]_{\mathcal{J}^c}\rangle - \langle A_{\mathcal{J}^c}^T \lambda, [x'-x_o]_{\mathcal{J}^c}\rangle\\
			&\geq \|x_o\|_1 + (1-\|A_{\mathcal{J}^c}^T \lambda\|_\infty) \|[x'-x_o]_{\mathcal{J}^c}\|_1
		\end{aligned}
	\end{equation}
	\item By contradiction, suppose $[x'-x_o]_{\mathcal{J}^c} = 0$. Then $A(x' - x_o) = 0 = A_\mathcal{J} [x'-x_o]_\mathcal{J}$. Thus $[x'-x_o]_\mathcal{J} = 0$ because $A_\mathcal{J}$ is full rank, and thus $x' = x_o$, which is a contradiction.
\end{enumerate}
\subsection{Constructing incoherent matrices}
\begin{thm}
	Let $v \backsim Uni(\mathcal{S}^{n-1})$, then $Pr[|v_n| \geq \varepsilon] \leq \frac{1}{n \varepsilon^2} \xrightarrow{n \to \infty} 0$
\end{thm}
We can prove this theorem by using the fact that $v$ is of norm 1: $\sum_{i=1}^{n} v_i^2 = 1$.\\
We can thus say that $\sum_{i=1}^{n} \mathbb{E}[v_i^2] = 1$, or equivalently $\mathbb{E}[v_n^2] = \frac{1}{n}$ because the $v_i$ are i.i.d.\\
Using the markov inequality $Pr[v_n^2 \geq a] \leq \frac{\mathbb{E}[v_n^2]}{a}$, it yields $Pr[|v_n| \geq \varepsilon] = Pr[v_n^2 \geq \varepsilon^2] \leq \frac{1}{n \varepsilon^2}$.\\
\begin{thm}
	Let $A = [a_1 | \dots | a_n]$, with $a_i \backsim Uni(\mathcal{S}^{m-1})$ i.i.d. Then, $\forall \delta > 0: \exists C_\delta, \forall m,n$ such that 
	\begin{equation}
		Pr\left[\mu(A) \geq C_\delta \sqrt{\frac{log(n)}{m}}\right] \geq 1 - \delta
	\end{equation}
	or equivalently 
	\begin{equation}
		\mu(A) \leq C \sqrt{\frac{log(n)}{m}} \qquad \text{with high probability.}
	\end{equation}
\end{thm}
Combining this result with the theorem \ref{thm:ell1_inco}, we get that, with high probability,
\begin{equation}
	\|x_o\|_0 \leq \frac{1}{2C} \sqrt{\frac{m}{log(n)}} 
\end{equation}
and we should be able to recover $x_o$ from $m \geq 4C^2k^2log(n)$ observations.
\subsection{Restricted isometry property}
\begin{definition}
	The matrix $A$ satisfies the \textbf{restricted isometry property (RIP)} of order $k$, with constant $\delta_k \in [0,1)$, if $\forall \|x\|_0 = k$, we have
	\begin{equation}
		(1-\delta_k)\|x\|_2^2 \leq \|Ax\|_2^2 \leq (1+\delta_k)\|x\|_2^2
	\end{equation}
\end{definition}
\begin{thm}[$\ell^0$ recovery under RIP]\label{thm:ell0_rip}
	Let $y = Ax_o$ with $k \coloneqq \ \|x_o\|_0$. If $\delta_{2k}(A) < 1$, then $x_o$ is the unique optimal solution to the $\ell^0$ minimization problem.
\end{thm}
Let us prove this theorem by contradiction. Let $x'$ be a solution different from $x_o$ to the $\ell^0$ minimization problem. Then the theorem yields 
\begin{equation}
	\begin{aligned}
		(1-\delta_{2k}) \|x'-x_o\|_2^2 &\leq \|A(x'-x_o)\|_2^2\\
	\end{aligned}
\end{equation}
But $A(x'-x_o) = 0$ because $Ax' = y = Ax_o$. Thus, $\|x'-x_o\|_2^2 \leq 0$, which is a contradiction because $x'\neq x_o$.\\
\begin{thm}[$\ell^1$ recovery under RIP]\label{thm:ell1_rip}
	Let $y = Ax_o$ with $k \coloneqq \|x_o\|_0$. If $\delta_{2k}(A) < \sqrt{2}-1$, then $x_o$ is the unique optimal solution to the $\ell^1$ minimization problem.
\end{thm}
\begin{thm}[RIP of Gaussian matrices]
	Let $A \in \R^{m \times n}$ be a matrix with i.i.d. entries following the probability repartition $\mathcal{N}(0,1/m)$. Then $\exists C > 0$ such that, with high probability, $\delta_k(A) < \delta$ provided that $m \geq C \frac{k}{\delta^2} \log(n/k)$.
\end{thm}
\begin{thm}
	Let $v_1,\dots,v_n\in \R^D$. Let $A\in \R^{m\times D}$ be a random matrix whose entries are independent random variables of probability distribution $\mathcal{N}(0,1/m)$. Then, for any $\varepsilon\in (0,1)$, with probability at least $1-1/n^2$, the following holds:
	\begin{equation}
		\forall i\neq j, \ (1-\varepsilon)\|v_i-v_j\|_2^2\le \|Av_i-Av_j\|_2^2 \le (1+\varepsilon)\|v_i-v_j\|_2^2
	\end{equation}
	provided that $m\ge 32\log(n)/\varepsilon^2$. 
\end{thm}
\subsection{Noisy observations or approximate sparsity}
First let us define the problem of noisy observations. We have $y = Ax_o + z$ with $\|z\|_2 \leq \epsilon$ for $\epsilon > 0$. The $\ell^1$ minimization problem in this case is:
\begin{equation}
	\min_{x} \|x\|_1 \quad \text{such that } \|y - Ax\|_2 \leq \epsilon.
\end{equation}
This is called the basis pursuit denoising (BPDN) problem. Now let us introduce the Lagrangian relaxation of this problem, also called the Lasso (Least Absolute Shrinkage and Selection Operator) problem:
\begin{equation}
	\min_x \lambda \|x\|_1 + \frac{1}{2}\|y - Ax\|_2^2 
\end{equation}
With noise, we cannot expect to recover $x_o$ exactly, but we can hope to recover an approximation of it. However, we can hope that if the noise level $\epsilon$ is small, then the estimation error $\|\hat{x}-x_o\|$ will also be small. We can bound this estimation error like this:
\begin{equation}
	\begin{aligned}
		A_\mathcal{J}(\hat{x} - x_o)_\mathcal{J} &= A_\mathcal{J} (A_\mathcal{J}^* A_\mathcal{J})^{-1} A_\mathcal{J}^* y - A_\mathcal{J} x_{o,\mathcal{J}}\\
		&= A_\mathcal{J} (A_\mathcal{J}^* A_\mathcal{J})^{-1} A_\mathcal{J}^* z
	\end{aligned}
\end{equation}
Hence:
\begin{equation}
	\begin{aligned}
		\sigma_{\min}(A_\mathcal{J}) \|(\hat{x} - x_o)_\mathcal{J}\|_2 &\leq \|A_\mathcal{J} (\hat{x} - x_o)_\mathcal{J}\|_2\\
		&\leq \| A_\mathcal{J} (A_\mathcal{J}^* A_\mathcal{J})^{-1} A_\mathcal{J}^* z\|_2
		&\leq \|z\|_2 \leq \epsilon
	\end{aligned}
\end{equation}
And so $\|\hat{x} - x_o\|_2 \leq \frac{\epsilon}{\sigma_{\min}(A_I)}$.\\
Because $x_o$ is feasible for the optimization problem and $\hat{x}$ is optimal, $\hat{x}$ must have a lower objective value than $x_o$:
\begin{equation}
	\|\hat{x}\|_1 \leq \|x_o\|_1
\end{equation}
Consider $h = \hat{x} - x_o$, the perturbation of $x_o$ that we want to bound. Let us define again $\mathcal{J} = supp(x_o)$ and $\mathcal{J}^c$ its complement. We have:
\begin{equation}
	\begin{aligned}
		\|x_o\|_1 &\geq \|\hat{x}\|_1 = \|x_o + h\|_1\\
		&\geq \|x_o\|_1 - \|h_\mathcal{J}\|_1 + \|h_{\mathcal{J}^c}\|_1 \\
		\|h_\mathcal{J}\|_1 &\geq \|h_{\mathcal{J}^c}\|_1
	\end{aligned}
\end{equation}
\begin{thm}
	Let $y = Ax_o + z$ with $\|z\|_2 \leq \epsilon$. Let $\hat{x}$ be the solution to the BPDN problem. Then for any $k$ such that $\delta_{2k}(A) < \sqrt{2}-1$, we have
	\begin{equation}
		\|\hat{x} - x_o\|_2 \leq C \frac{\|x_o - [x_o]_k\|_1}{\sqrt{k}} + C' \epsilon
	\end{equation}
	for some constants $C$ and $C'$ which only depend on $\delta_{2k}(A)$.
\end{thm}
\section{Convex methods for Low-rank matrix recovery}
Now we want to interest ourselves in the recovery of a low-rank matrix $X_o\in \R^{n_1\times n_2}$ based on linear measurements $y = \mathcal{A}[X_o]$. We thus want that $rank(X_o) = \|\boldsymbol{\sigma}(X_o)\|_0 \leq r$. From this, we define the following optimization problems which is similar to the ones we have for sparse recovery:\\
The $(\boldsymbol{\sigma}-\ell^0)$ minimization problem:
\begin{equation}
	\min_{X\in \R^{n_1\times n_2}} \|\boldsymbol{\sigma}(X)\|_0 \qquad \text{ such that } y = \mathcal{A}[X]
\end{equation}
The $(\boldsymbol{\sigma}-\ell^1)$ minimization problem:
\begin{equation}
	\min_{X\in \R^{n_1\times n_2}} \|\boldsymbol{\sigma}(X)\|_1 \qquad \text{ such that } y = \mathcal{A}[X]
\end{equation}
We can reformulate these problem as:
\begin{equation}
	\min_{X\in \R^{n_1\times n_2}} \|\mathcal{A}[X]-y\|^2 \qquad \text{ such that } rank(X) \leq r
\end{equation}
\subsection{Representing low-rank matrices}
We have seen in the first part of the course that we can use manifold optimization to solve the problem of low-rank matrix recovery. For that, we can explore the representation of low-rank matrices through manifolds such as the Stiefel manifold.
\begin{equation}
	\begin{aligned}
		rank(X) \leq r &\Longleftrightarrow \exists &&U\in St(n_1,r) = \{Y \in \R^{n_1 \times r} | Y^T Y = I_r\},\\
		& &&V\in St(n_2,r), \Sigma \in \R^{r \times r} \text{ diagonal}\\
		& &&\text{ such that } X = U \Sigma V^T\\
		&\Longleftrightarrow \exists && L \in \R^{n_1 \times r}, R \in \R^{n_2 \times r} \text{ such that } X = LR^T\\
		&\Longleftrightarrow \exists &&L \in St(n_1,r), R \in \R^{n_2 \times r} \text{ such that } X = LR^T\\
		&\Longleftrightarrow \exists &&U \in St(n_1,r), V \in St(n_2,r), S\in \R^{r \times r} \text{ such that } X= USV^T\\
		&\Longleftrightarrow \exists &&L \in \R^{(n_1-r) \times r} \text{ full rank such that } LX = 0\\
		&\Longleftrightarrow \exists &&R \in \R^{(n_2-r) \times r} \text{ full rank such that } XR = 0
	\end{aligned}
\end{equation}
\subsection{Recovering a low-rank matrix}
Reminder we want to find $X_o$ such that $y = \mathcal{A}[X_o]$ and $rank(X_o) \leq r$. For that we want to solve the problem:
\begin{equation}
	\min_{X\in \R^{n_1\times n_2}} rank(X) \qquad \text{ such that } y = \mathcal{A}[X]
\end{equation}
We saw in the start of the section that we could relax this problem by replacing the rank minimization by the nuclear norm minimization, because the nuclear norm is the convex envelope of the rank. But how can we guarantee that the solution of the nuclear norm minimization problem is the same as the one of the rank minimization problem? For that, we need to introduce a property of $\mathcal{A}$ called the rank-RIP (Restricted Isometry Property).
\begin{definition}
	$\mathcal{A}$ has rank-RIP of rank $r$ with constant $\delta$ if $\exists \gamma > 0 $ such that $\forall X\in \R^{n_1\times n_2}$ with $rank(X) \leq r$, we have
	\begin{equation}
		(1-\delta)\|X\|_F^2 \leq \gamma\|\mathcal{A}[X]\|_2^2 \leq (1+\delta)\|X\|_F^2
	\end{equation}
	The rank-$r$ RIP constant $\delta_r(\mathcal{A})$ is the smallest $\delta$ such that the above inequality holds.
\end{definition}
he rank-RIP implies uniqueness of structured (low-rank) solutions:
\begin{thm}
	If $y = \mathcal{A}[X_o]$, $rank(X_o) \leq r$ and $\delta_{2r}(\mathcal{A}) < 1$, then $X_o$ is the unique solution to the $(\boldsymbol{\sigma}-\ell^0)$ minimization problem.
\end{thm}
The existence and the uniqueness of this solution can be proven:
\begin{itemize}
	\item \textbf{Existence}: the range of the objective function is finite so there exists a solution to the problem.
	\item \textbf{Uniqueness}: Let $\hat{X}$ be a solution different from $X_o$. Then, using the rank-RIP property, we have
	\begin{equation}
		(1-\delta_{2r}) \|\hat{X}-X_o\|_F^2 \leq \gamma \|\mathcal{A}[\hat{X}-X_o]\|_2^2
	\end{equation}
	The left term is positive because $(1-\delta_{2r}) > 0$ and norm is positive definite, and the right term is zero because $\mathcal{A}[\hat{X}-X_o] = \mathcal{A}[\hat{X}] - \mathcal{A}[X_o] = y-y=0$. This is a contradiction, thus $\hat{X}$ cannot be different from $X_o$.  
\end{itemize} 
\begin{thm}
	Let $y = \mathcal{A}[X_o]$, $rank(X_o) \leq r$ and $\delta_{2r}(\mathcal{A}) < \sqrt{2}-1$, then $X_o$ is the unique solution to the $(\boldsymbol{\sigma}-\ell^1)$ minimization problem. 
\end{thm}
\subsection{Low-rank matrix completion}
The problem of low-rank matrix completion can be define as:\\
Let $X_o \in \R^{n_1 \times n_2}$ be a low-rank matrix. We are given $(X_o)_{\Omega}$ where $\Omega \subset [n_1]\times [n_2]$ is a subset of the entries $\Omega^c$ of $X_o$. We want to fill the missing entries of $X_o$ by using the low-rank property of $X_o$.\\
Unilke the previous problems, rank-RIP cannot hold to form such an $\mathcal{A}$. So we need other way to solve this problem, for that, we can use the nuclear norm and define the following optimization problem:
\begin{equation}
	\min_{X\in \R^{n_1 \times n_2}} \|X\|_* \qquad \text{ such that } X_\Omega = y
\end{equation}
With $y$ the observed entries of $X_o$.\\
With this problem, we can recover $X_o$, but what if there are some noise in the observed entries $Y_{i,j} = [X_o]_{i,j} + z_{i,j}$ for $(i,j) \in \Omega$ and $\|P_\Omega(z)\|_F \leq \epsilon$ ? Then the optimization problem becomes:
\begin{equation}
	\min_{X\in \R^{n_1 \times n_2}} \|X\|_* \qquad \text{ such that } \|P_\Omega(X)-P_\Omega(y)\|_F \leq \epsilon
\end{equation} 
\subsection{Manifold of fixed-rank matrices}
\begin{definition}
	The manifold of fixed-rank matrices is the set of matrices of rank $r$, it is defined as:
	\begin{equation}
		\R_r^{n_1 \times n_2} = \{X\in \R^{n_1\times n_2} | rank(X) = r\}
	\end{equation}
\end{definition}
\begin{thm}
	The manifold of fixed-rank matrices $\R_r^{n_1 \times n_2}$ is a smooth embedded submanifold of $\R^{n_1 \times n_2}$.
\end{thm}
